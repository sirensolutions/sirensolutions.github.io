# Siren NLP
The Siren NLP plugin provides an Elasticsearch ingestion pipeline with a variety of processors that enriches documents by performing entity extraction.

Entity extraction is a technique that identifies and segments the named entities and classifies or categorizes them under various predefined classes.

You can use the NLP plugin to enrich text fields with annotations for named entities, such as Organization, Person, or Location, and predefined taxonomies.

For more details, see our [documentation](https://docs.support.siren.io/siren-nlp/0.5.1/siren-nlp/introduction.html).

<script>
	document.addEventListener("DOMContentLoaded", function () {
		document.querySelectorAll(".paginated-table").forEach(function (wrapper) {
			var tableBody = wrapper.querySelector("tbody");
			var controls = wrapper.querySelector(".table-pagination");

			if (!tableBody || !controls) {
				return;
			}

			var rows = Array.prototype.slice.call(tableBody.querySelectorAll("tr"));
			var pageSize = parseInt(wrapper.dataset.pageSize || "20", 10);
			var pageCount = Math.ceil(rows.length / pageSize);

			if (pageCount <= 1) {
				controls.hidden = true;
				return;
			}

			function renderPage(pageIndex) {
				var start = pageIndex * pageSize;
				var end = start + pageSize;

				rows.forEach(function (row, index) {
					row.hidden = index < start || index >= end;
				});

				Array.prototype.forEach.call(controls.querySelectorAll("button[data-page]"), function (button) {
					button.classList.toggle("is-active", Number(button.dataset.page) === pageIndex);
					button.setAttribute("aria-current", Number(button.dataset.page) === pageIndex ? "page" : "false");
				});
			}

			for (var pageIndex = 0; pageIndex < pageCount; pageIndex += 1) {
				var button = document.createElement("button");
				button.type = "button";
				button.dataset.page = pageIndex;
				button.textContent = String(pageIndex + 1);
				button.addEventListener("click", function (event) {
					renderPage(Number(event.currentTarget.dataset.page));
				});
				controls.appendChild(button);
			}

			renderPage(0);
		});
	});
</script>
